from collections import deque
result, tempArray = [9, 7, 5, 3, 1, 6, 2, 4, 8], []
result, tempArray = deque(result), deque(tempArray)
while len(result) != 0:
    tempArray.append(result.pop())
    print("加入数据", tempArray)
    if len(result) > 0:
        tempArray.append(tempArray.popleft())
        print('结果', tempArray)
