## 用栈来求解汉诺塔问题

* 题目:

* 难度:☆☆☆

* 代码实现:

```
count = 0
towerA = {"A": []}
towerB = {"B": []}
towerC = {"C": []}


def PushToTower(n):
    for i in range(n, 0, -1):
        towerA["A"].append(i)


def moveAction(From, To):
    global count
    FromName = list(From.keys())[0]
    ToName = list(To.keys())[0]

    popValue = From[FromName].pop()
    To[ToName].append(popValue)
    count += 1
    return print("A柱:" + str(towerA["A"]) + "\nB柱:" + str(towerB["B"]) + "\nC柱:" + str(towerC["C"]) + "\n步骤描述:%s从%s移动到%s" % (popValue, FromName, ToName) + "\n第%s步" % count + "\n-------------------------------------------------------")


# f b t形参其中的含义主要是指，将对应tower数组里面pop出一个数，做为操纵对象，操作到指定tower中去，例如 TowerA["A"]数组中POP一个最顶上的数，push到TowerC中去，其中要求必须经过TowerB
def moveitem(num, f, b, t):
    if 1 == num:
        moveAction(f, b)  # 最后倒数第二个步骤把1从A移动到B
        moveAction(b, t)  # 最后倒数第一个步骤把1从B移动到C
        return

    # 最大一个要从左A移动到右C有两个关键步骤。从A到B，然后从B到C，从A到B 之前需要先将顶上部分n-1移动到C，也就是一下过程：
    moveitem(num - 1, f, b, t)
    # n-1部分已经移动到C了，一下步骤是把最大一个从A移动到B
    moveAction(f, b)
    # 最大一个已经从A移动到B了，以下步骤又要将n-1部分从C移动到A，因为需要把C的位置空出来放最大的
    moveitem(num - 1, t, b, f)
    # n-1部分已经从C移动到A了，以下步骤是把最大一个从B移动到C
    moveAction(b, t)
    # 以下步骤是将n-1部分放到C最大一个的上方完成一次递归
    moveitem(num - 1, f, b, t)


def startMove(n):
    PushToTower(n)
    moveitem(n, towerA, towerB, towerC)


startMove(3)
print("共计执行步骤:", count)
```
