## 反转部分单向、双向链表
* 题目：略
* 要求：1.如果链表长度为N，时间复杂度为O（N），空间复杂度为O（1）
* 难度：☆
* 代码实现:
```
class Node(object):
    def __init__(self, data, pnext = None, ppre = None):
        self.data = data
        self._next = pnext
        self._pre = ppre

    def __repr__(self):
        return str(self.data)

class ChainTable(object):
    def __init__(self):
        self.head = None
        self.length = 0

    def isEmpty(self):
        return (self.length == 0)

    def append(self, dataOrNode):
        item = None
        if isinstance(dataOrNode, Node):
            item = dataOrNode
        else:
            item = Node(dataOrNode)

        if not self.head:
            self.head = item
            self.length += 1

        else:
            node = self.head
            while node._next != None:
                node = node._next
            node._next = item
            item._pre = node
            self.length += 1

def reverseChain(chain):
    tempNode = chain.head
    for i in range(chain.length):
        if tempNode._next != None:
            tempPre = tempNode._pre 
            tempNext = tempNode._next
            tempNode._next = tempPre
            tempNode._pre = tempNext
            tempNode = tempNode._pre

            chain.head = tempNode
            tempPre = tempNode._pre 
            tempNext = tempNode._next
            tempNode._next = tempPre
            tempNode._pre = tempNext
    return chain
```
